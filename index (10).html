⚙️ Conectar con Google Sheets

Para que los pedidos lleguen a tu planilla, necesitás crear un pequeño script en Google. Seguí estos pasos:

1. Abrí tu Google Sheets → Extensiones → Apps Script
2. Borrá todo el contenido y pegá el código de abajo
3. Hacé clic en Implementar → Nueva implementación
4. Tipo: App web | Acceso: Cualquier persona
5. Copiá la URL y pegala aquí:

Código para Apps Script:

var CORS_HEADERS = { "Access-Control-Allow-Origin": "*", "Access-Control-Allow-Methods": "GET, POST", "Access-Control-Allow-Headers": "Content-Type" }; function doGet(e) { try { var ss = SpreadsheetApp.getActiveSpreadsheet(); var telas = []; var colores = []; var sheetTelas = ss.getSheetByName("Telas"); if (sheetTelas && sheetTelas.getLastRow() > 1) { var vt = sheetTelas.getRange("A2:A" + sheetTelas.getLastRow()).getValues(); telas = vt.map(function(r){ return r[0]; }).filter(function(v){ return v !== ""; }); } var sheetColores = ss.getSheetByName("COLORES"); if (sheetColores && sheetColores.getLastRow() > 1) { var vc = sheetColores.getRange("A2:A" + sheetColores.getLastRow()).getValues(); colores = vc.map(function(r){ return r[0]; }).filter(function(v){ return v !== ""; }); } return ContentService .createTextOutput(JSON.stringify({ok: true, telas: telas, colores: colores})) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService .createTextOutput(JSON.stringify({ok: false, error: err.toString()})) .setMimeType(ContentService.MimeType.JSON); } } function doPost(e) { try { var data = JSON.parse(e.postData.contents); var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName("Pedidos") || ss.insertSheet("Pedidos"); if (sheet.getLastRow() === 0) { sheet.appendRow(["Fecha","Vendedor","Cliente","N° Pedido", "Tela #","Nombre Tela","N° Orden","Pagado","Tipo","Subtipo Cabezal","Unidad Corte", "Colorido","Cantidad","Dirección","Comentarios"]); } var fecha = new Date().toLocaleString("es-AR"); var pedidos = data.pedidos || [data]; pedidos.forEach(function(pedido) { pedido.telas.forEach(function(t, i) { sheet.appendRow([fecha, pedido.vendedor, pedido.cliente, pedido.nroPedido||"", i+1, t.nombreTela, t.nroOrden||"", t.pago||"", t.tipo, t.subtipo||"", t.unidad||"", t.colorido||"", t.cantidad||"", pedido.direccion||"", pedido.comentarios||""]); }); }); return ContentService .createTextOutput(JSON.stringify({ok: true})) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService .createTextOutput(JSON.stringify({ok: false, error: err.toString()})) .setMimeType(ContentService.MimeType.JSON); } }
