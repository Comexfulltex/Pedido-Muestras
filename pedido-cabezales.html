⚙️ Conectar con Google Sheets

Para que los pedidos lleguen a tu planilla, necesitás crear un pequeño script en Google. Seguí estos pasos:

1. Abrí tu Google Sheets → Extensiones → Apps Script
2. Borrá todo el contenido y pegá el código de abajo
3. Hacé clic en Implementar → Nueva implementación
4. Tipo: App web | Acceso: Cualquier persona
5. Copiá la URL y pegala aquí:

Código para Apps Script:

function doPost(e) { try { var data = JSON.parse(e.postData.contents); var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName("Pedidos") || ss.insertSheet("Pedidos"); if (sheet.getLastRow() === 0) { sheet.appendRow(["Fecha","Vendedor","Cliente", "N° Pedido","Tela #","Nombre Tela","Tipo", "Colorido","Dirección","Comentarios"]); } var fecha = new Date().toLocaleString("es-AR"); data.telas.forEach(function(t, i) { sheet.appendRow([fecha, data.vendedor, data.cliente, data.nroPedido, i+1, t.nombreTela, t.tipo, t.colorido, data.direccion||"", data.comentarios||""]); }); return ContentService .createTextOutput(JSON.stringify({ok:true})) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService .createTextOutput(JSON.stringify({ok:false,error:err.toString()})) .setMimeType(ContentService.MimeType.JSON); } }
